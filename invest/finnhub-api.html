1. 取得 API Key

  1. 到 finnhub.io/register 註冊（免費）
  2. 登入後到 Dashboard，複製 API Key
  3. 放進環境變數（不要寫進程式）

2. 安裝官方 Python SDK

# pip 安裝
pip install finnhub-python

# 或升級到最新版
pip install --upgrade finnhub-python

3. 初始化客戶端

import finnhub
import os

# 從環境變數讀 API key（推薦做法）
finnhub_client = finnhub.Client(api_key=os.getenv("FINNHUB_API_KEY"))

# 或測試時直接傳
# finnhub_client = finnhub.Client(api_key="YOUR API KEY")

# 簡易測試
print(finnhub_client.quote("AAPL"))

4. 純 REST（不裝 SDK）

# 用 curl 也可以,base URL 是 https://finnhub.io/api/v1
curl "https://finnhub.io/api/v1/quote?symbol=AAPL&token=YOUR_API_KEY"

# 回傳 JSON
{
  "c": 189.84,   // current price
  "d": -1.23,    // change
  "dp": -0.6437, // percent change
  "h": 191.55,  // high price of the day
  "l": 189.20,  // low price of the day
  "o": 191.07,  // open price of the day
  "pc": 191.07, // previous close
  "t": 1721054400  // timestamp
}
Base URL: https://finnhub.io/api/v1
認證方式: URL 帶 token=YOUR_KEY query 參數（不是 Bearer header）
回傳格式: 純 JSON，沒有標準錯誤 envelope，錯的時候回空物件或 {}

範例 3：基本面 + 分析師目標價綜合評分

def analyze_fundamentals(symbol):
    client = finnhub.Client(api_key=os.getenv("FINNHUB_API_KEY"))

    # 1. 基本財務指標
    bs = client.company_basic_financials(symbol, "all")["metric"]
    pe = bs.get("peNormalizedAnnual")
    roe = bs.get("roeRfy")
    gross_margin = bs.get("grossMarginAnnual")

    # 2. 分析師目標價
    pt = client.price_target(symbol)
    target_avg = pt["targetMean"]
    target_high = pt["targetHigh"]
    target_low = pt["targetLow"]

    # 3. 即時報價
    q = client.quote(symbol)
    current = q["c"]

    upside = (target_avg - current) / current * 100

    print(f"""
=== {symbol} 基本面分析 ===
現價: ${current:.2f}
PE: {pe:.1f}
ROE: {roe*100:.1f}%
毛利率: {gross_margin*100:.1f}%

分析師目標價:
  平均: ${target_avg:.2f} ({upside:+.1f}% upside)
  最高: ${target_high:.2f}
  最低: ${target_low:.2f}
""")

    return {
        "symbol": symbol,
        "current": current,
        "pe": pe,
        "upside": upside,
        "analyst_target": target_avg,
    }

analyze_fundamentals("AAPL")

範例 4：法說會逐字稿 NLP — 管理層信心指標

import finnhub
import re

client = finnhub.Client(api_key=os.getenv("FINNHUB_API_KEY"))

# 1. 列出 AAPL 過去 4 季法說會
transcripts = client.transcripts_list("AAPL")

# 信心詞 / 模糊詞 清單
confident_words = ["strong", "confident", "robust", "exceeded", "outperform", "record"]
hedge_words = ["challenging", "headwind", "uncertain", "may", "could", "approximately"]

for t in transcripts[:4]:
    # 2. 抓該場逐字稿
    full = client.transcripts(t["id"])
    text = full.get("transcript", "").lower()

    # 3. 簡單信心分數
    confident = sum(text.count(w) for w in confident_words)
    hedge = sum(text.count(w) for w in hedge_words)
    score = (confident - hedge) / max(confident + hedge, 1) * 100

    print(f"{t['quarter']} {t['year']}: 信心={score:+.0f} ({confident} 強 / {hedge} 弱)")
💡 進階用法：可以把逐字稿餵給 Claude / GPT 做 sentiment analysis，比單純數字高/低更精準。

⑥ WebSocket 即時串流

Finnhub 的 WebSocket 是 Pro 以上方案才能用的即時報價（Free 是延遲版）。端點：wss://ws.finnhub.io

安裝 websocket-client

pip install websocket-client

Python 範例：訂閱 AAPL 即時 tick

import websocket
import json
import os

def on_message(ws, message):
    data = json.loads(message)
    print(data)

def on_open(ws):
    # 訂閱 AAPL
    ws.send(json.dumps({"type": "subscribe", "symbol": "AAPL"}))

def on_close(ws):
    print("連線關閉")

ws = websocket.WebSocketApp(
    "wss://ws.finnhub.io?token=" + os.getenv("FINNHUB_API_KEY"),
    on_message=on_message,
    on_open=on_open,
    on_close=on_close
)
ws.run_forever()

⑧ 常見坑

坑 1：時間戳是 Unix 秒，不是毫秒

# ❌ 錯誤：傳毫秒 timestamp
res = client.stock_candles("AAPL", "D", 1721054400000, 1721054400000)
# Finnhub 不會報錯,但會回空資料 {s: "no_data"}

# ✅ 正確：傳秒
res = client.stock_candles("AAPL", "D", 1721054400, 1721054400)

坑 2：免費方案的 _from/_to 範圍太近會拿到空資料

# 假設今天是 2026-07-11,免費方案只能看最近 15 天分線
# _from 設定成 30 天前會拿到空 dict

# ✅ 解決：先檢查 res["s"] 狀態
res = client.stock_candles("AAPL", "1", _from, to)
if res["s"] == "no_data":
    print("此時間範圍無資料,可能是方案限制")

坑 3：API Key 不要寫在程式碼

# ❌ 危險
client = finnhub.Client(api_key="c123abc456")

# ✅ 用環境變數
# export FINNHUB_API_KEY=c123abc456
client = finnhub.Client(api_key=os.getenv("FINNHUB_API_KEY"))

坑 4：Rate Limit 沒處理會被 ban

import time
from functools import wraps

def rate_limit(max_per_min=60):
    interval = 60 / max_per_min
    def decorator(fn):
        @wraps(fn)
        def wrapper(*args, **kwargs):
            time.sleep(interval)
            return fn(*args, **kwargs)
        return wrapper
    return decorator

# 套用在呼叫上
@rate_limit(max_per_min=50)  # 留 10 個 buffer
def fetch(symbol):
    return client.quote(symbol)

坑 6：錯誤回傳不像一般 REST 標準
